7   package com.codeferm.dbaccess.dbcp;
8   
9   import com.codeferm.dbaccess.BaseTest;
10  import java.io.IOException;
11  import java.sql.SQLException;
12  import javax.sql.DataSource;
13  import org.apache.commons.dbcp2.BasicDataSource;
14  import org.junit.After;
15  import org.junit.AfterClass;
16  import org.junit.Before;
17  import org.junit.BeforeClass;
18  
19  /**
20   * Base test class with common methods used by all other DBCP based tests.
21   *
22   * @author sgoldsmith
23   * @version 1.0.0
24   * @since 1.0.0
25   */
26  public class BaseDbcpTest extends BaseTest {
27  
28      /**
29       * Database {@code DataSource}.
30       */
31      private static DataSource dataSource;
32  
33      /**
34       * Get {@code BasicDataSource}.
35       *
36       * @return DataSource DBCP DataSource.
37       */
38      public static DataSource getDataSource() {
39          return dataSource;
40      }
41  
42      /**
43       * Initialize class.
44       *
45       * @throws IOException Possible exception.
46       */
47      @BeforeClass
48      public static void setUpClass() throws IOException {
49          // Call super setUpClass
50          BaseTest.setUpClass();
51          // Create DBCP DataSource
52          BasicDataSource ds = new BasicDataSource();
53          ds.setDriverClassName(getProperties().getProperty("db.driver"));
54          ds.setUsername(getProperties().getProperty("db.user"));
55          ds.setPassword(getProperties().getProperty("db.password"));
56          ds.setUrl(getProperties().getProperty("db.url"));
57          ds.setMaxTotal(Integer.parseInt(getProperties().getProperty(
58                  "db.pool.size")));
59          dataSource = ds;
60      }
61  
62      /**
63       * Close {@code DataSource}.
64       *
65       * @throws SQLException Possible exception.
66       */
67      @AfterClass
68      public static void tearDownClass() throws SQLException {
69          ((BasicDataSource) dataSource).close();
70      }
71  
72      /**
73       * Do nothing for now.
74       */
75      @Before
76      public void setUp() {
77      }
78  
79      /**
80       * Do nothing for now.
81       */
82      @After
83      public void tearDown() {
84      }
85  }
