26      /**
27       * Test of commit transaction using DbJdbcDs implementation.
28       *
29       * @throws Exception Possible exception.
30       */
31      @Test
32      public void commit() throws Exception {
33          final int loops = Integer.parseInt(getProperties().getProperty(
34                  "xa.pool.size")) * LOOP_MULTIPLIER;
35          log.info("commit DbJdbcDs prime");
36          long startTime = System.currentTimeMillis();
37          // Run one time to force any lazy loading, so elapse time is more
38          // accurate.
39          TransactionTest test = TransactionFactory.createObject(
40                  TransactionTest.class, AtomikosTransModule.class);
41          int id = test.commitRec(new DbJdbcDs(getDataSource()), getSqlMap().get(
42                  "insert.testtable"));
43          test.verifyCommit(new DbJdbcDs(getDataSource()), getSqlMap().get(
44                  "select.testtable.by.id"), getSqlMap().get(
45                          "delete.testtable.by.id"), id);
46          long elapsedTime = System.currentTimeMillis() - startTime;
47          log.info(String.format("Elapsed time: %d ms", elapsedTime));
48          log.info("commit DbJdbcDs");
49          startTime = System.currentTimeMillis();
50          for (int number = 1; number <= loops; number++) {
51              id = test.commitRec(new DbJdbcDs(getDataSource()), getSqlMap().get(
52                      "insert.testtable"));
53              test.verifyCommit(new DbJdbcDs(getDataSource()), getSqlMap().get(
54                      "select.testtable.by.id"), getSqlMap().get(
55                              "delete.testtable.by.id"), id);
56          }
57          elapsedTime = System.currentTimeMillis() - startTime;
58          log.info(String.format("Elapsed time: %d ms, loops: %d, average: %d ms",
59                  elapsedTime, loops, elapsedTime / loops));
60      }
