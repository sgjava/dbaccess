90      /**
91       * Return query results as list of beans. {@code Connection} is not closed.
92       *
93       * @param <T> Type of object that the handler returns
94       * @param sql SQL statement to execute
95       * @param params Initialize the PreparedStatement's IN parameters
96       * @param clazz Class to map results to
97       * @return {@code List} of {@code <T>} typed objects
98       */
99      @Override
100     @SuppressFBWarnings(value
101             = "SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING",
102             justification
103             = "SQL libraries are allowed to accept SQL as parameter")
104     public final <T> List<T> selectList(final String sql,
105             final Object[] params, final Class clazz) {
106         if (log.isDebugEnabled()) {
107             log.debug(String.format("selectBeanList: sql=%s, params=%s", sql,
108                     Arrays.asList(params)));
109         }
110         PreparedStatement preparedStatement = null;
111         ResultSet resultSet = null;
112         List<T> list = null;
113         try {
114             preparedStatement = connection.prepareStatement(sql);
115             // Fill parameters
116             int i = 1; //NOPMD
117             for (Object o : params) {
118                 preparedStatement.setObject(i++, o);
119             }
120             resultSet = preparedStatement.executeQuery();
121             list = createObjects(resultSet, clazz);
122         } catch (Exception e) {
123             throw new DbAccessException(String.format(
124                     "selectBeanList: sql=%s, params=%s", sql, Arrays.asList(
125                             params)), e);
126         } finally {
127             try {
128                 if (resultSet != null) {
129                     resultSet.close();
130                 }
131             } catch (SQLException e) {
132                 log.error(e.getMessage());
133             }
134             try {
135                 if (preparedStatement != null) {
136                     preparedStatement.close();
137                 }
138             } catch (SQLException e) {
139                 log.error(e.getMessage());
140             }
141         }
142         return list;
143     }

159         if (log.isDebugEnabled()) {
160             log.debug(String.format("selectMapList: sql=%s, params=%s", sql,
161                     Arrays.asList(params)));
162         }
163         PreparedStatement preparedStatement = null;
164         ResultSet resultSet = null;
165         List<Map<String, Object>> list = null;
166         try {
167             preparedStatement = connection.prepareStatement(sql);
168             // Fill parameters
169             int i = 1; //NOPMD
170             for (Object o : params) {
171                 preparedStatement.setObject(i++, o);
172             }
173             resultSet = preparedStatement.executeQuery();
174             final ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
175             final int numberOfColumns = resultSetMetaData.getColumnCount();
176             list = new ArrayList<Map<String, Object>>();
177             final Map map = new HashMap();
178             while (resultSet.next()) {
179                 for (i = 1; i <= numberOfColumns; i++) {
180                     map.put(resultSetMetaData.getColumnName(i), resultSet.
181                             getObject(i));
182                 }
183                 list.add(map);
184             }
185         } catch (SQLException e) {
186             throw new DbAccessException(String.format(
187                     "selectMapList: sql=%s, params=%s", sql, Arrays.asList(
188                             params)), e);
189         } finally {
190             try {
191                 if (resultSet != null) {
192                     resultSet.close();
193                 }
194             } catch (SQLException e) {
195                 log.error(e.getMessage());
196             }
197             try {
198                 if (preparedStatement != null) {
199                     preparedStatement.close();
200                 }
201             } catch (SQLException e) {
202                 log.error(e.getMessage());
203             }
204         }
205         return list;

252     /**
253      * Executes the given INSERT statement with parameter array and returns auto
254      * generate key. JDBC driver needs to support RETURN_GENERATED_KEYS.
255      * {@code Connection} is not closed.
256      *
257      * @param sql SQL statement to execute
258      * @param params Initialize the PreparedStatement's IN parameters
259      * @return Field name/value pairs of keys
260      */
261     @Override
262     @SuppressFBWarnings(value
263             = "SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING",
264             justification
265             = "SQL libraries are allowed to accept SQL as parameter")
266     public final Map<String, Object> updateReturnKeys(final String sql,
267             final Object[] params) {
268         if (log.isDebugEnabled()) {
269             log.debug(String.format("updateReturnKeys: sql=%s, params=%s", sql,
270                     Arrays.asList(params)));
271         }
272         PreparedStatement preparedStatement = null;
273         ResultSet resultSet = null;
274         Map<String, Object> keys = null;
275         try {
276             preparedStatement = connection.prepareStatement(sql,
277                     Statement.RETURN_GENERATED_KEYS);
278             // Fill parameters
279             int i = 1; //NOPMD
280             for (Object o : params) {
281                 preparedStatement.setObject(i++, o);
282             }
283             preparedStatement.executeUpdate();
284             resultSet = preparedStatement.getGeneratedKeys();
285             final ResultSetMetaData resultSetMetaData = resultSet.getMetaData();
286             final int numberOfColumns = resultSetMetaData.getColumnCount();
287             // Prime the pump
288             if (resultSet.next()) {
289                 keys = new HashMap<String, Object>();
290                 for (i = 1; i <= numberOfColumns; i++) {
291                     keys.put(resultSetMetaData.getColumnName(i),
292                             resultSet.getObject(i));
293                 }
294             }
295         } catch (SQLException e) {
296             throw new DbAccessException(String.format(
297                     "updateReturnKeys: sql=%s, params=%s", sql, Arrays.asList(
298                             params)), e);
299         } finally {
300             try {
301                 if (resultSet != null) {
302                     resultSet.close();
303                 }
304             } catch (SQLException e) {
305                 log.error(e.getMessage());
306             }
307             try {
308                 if (preparedStatement != null) {
309                     preparedStatement.close();
310                 }
311             } catch (SQLException e) {
312                 log.error(e.getMessage());
313             }
314         }
315         return keys;
316     }

331     public final int[] batch(final String sql, final Object[][] params) {
332         if (log.isDebugEnabled()) {
333             log.debug(String.format("batch: sql=%s", sql));
334             for (Object[] param : params) {
335                 log.debug(
336                         String.format("batch: params=%s",
337                                 Arrays.asList(param)));
338             }
339         }
340         PreparedStatement preparedStatement = null;
341         int[] rows = null;
342         try {
343             preparedStatement = connection.prepareStatement(sql);
344             // Add parameter arrays to batch
345             for (Object[] param : params) {
346                 // Fill parameters
347                 int i = 1; //NOPMD
348                 for (Object o : param) {
349                     preparedStatement.setObject(i++, o);
350                 }
351                 // Add to batch
352                 preparedStatement.addBatch();
353             }
354             rows = preparedStatement.executeBatch();
355         } catch (SQLException e) {
356             throw new DbAccessException(String.format(
357                     "batch: sql=%s, params=%s", sql, Arrays.asList(
358                             params)), e);
359         } finally {
360             try {
361                 if (preparedStatement != null) {
362                     preparedStatement.close();
363                 }
364             } catch (SQLException e) {
365                 log.error(e.getMessage());
366             }
367         }
368         return rows;
369     }
