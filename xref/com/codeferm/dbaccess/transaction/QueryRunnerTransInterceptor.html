9   import com.codeferm.dbaccess.DbQueryRunnerConn;
10  import java.lang.reflect.Method;
11  import org.aopalliance.intercept.MethodInterceptor;
12  import org.aopalliance.intercept.MethodInvocation;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;

16  /**
17   * Intercept methods annotated with
18   * {@link com.codeferm.dbaccess.transaction.Transaction} and commit on success
19   * or rollback if {@link java.lang.Exception} is thrown. The first method
20   * parameter must be an instance of {@link com.codeferm.dbaccess.DbAccess} in
21   * order for this class to access the connection prior to commit or rollback
22   * operation as required by JDBC style transactions.
23   * <p>
24   * <p>
25   * Currently this interceptor supports only
26   * {@link com.codeferm.dbaccess.DbQueryRunnerConn}. Other {@code Connection}
27   * based implementations of {@link com.codeferm.dbaccess.DbAccess} would work as
28   * long as the getConnection method is implemented.
29   * <p>
30   * <p>
31   * Limitations
32   * <p>
33   * Behind the scenes, method interception is implemented by generating bytecode
34   * at runtime. Guice dynamically creates a subclass that applies interceptors by
35   * overriding methods. If you are on a platform that doesn't support bytecode
36   * generation (such as Android), you should use Guice without AOP support.
37   * <p>
38   * This approach imposes limits on what classes and methods can be intercepted:
39   * <p>
40   * <p>
41   * Classes must be public or package-private. Classes must be non-final Methods
42   * must be public, package-private or protected Methods must be non-final
43   * Instances must be created by Guice by an @Inject-annotated or no-argument
44   * constructor
45   * <p>
46   * <p>
47   * It is not possible to use method interception on instances that aren't
48   * constructed by Guice.
49   *
50   * @see com.codeferm.dbaccess.transaction.Transaction
51   * @see com.codeferm.dbaccess.transaction.QueryRunnerTransModule
52   * @see com.codeferm.dbaccess.transaction.TransactionFactory
53   *
54   * @author sgoldsmith
55   * @version 1.0.0
56   * @since 1.0.0
57   */
58  public final class QueryRunnerTransInterceptor implements MethodInterceptor {
59  
60      /**
61       * Logger.
62       */
63      //CHECKSTYLE:OFF ConstantName - Logger is static final, not a constant
64      private static final Logger log = LoggerFactory.getLogger(//NOPMD
65              QueryRunnerTransInterceptor.class);
66      //CHECKSTYLE:ON ConstantName
67  
68      /**
69       * Invoke method wrapped in a transaction.
70       *
71       * @param invocation Method invocation.
72       * @return Invoked Object.
73       * @throws Throwable Possible exception.
74       */
75      @Override
76      public Object invoke(final MethodInvocation invocation) throws Throwable {
77          final Method method = invocation.getMethod();
78          final Transaction annotation = method.getAnnotation(Transaction.class);
79          Object object = null;
80          // Make sure interceptor was called for a @Transaction method
81          if (annotation == null) {
82              // Not a Transaction annotated method, so proceed without change
83              object = invocation.proceed();
84  
85          } else {
86              // First parameter must be DbAccess object
87              if (invocation.getArguments().length == 0
88                      || !(invocation.getArguments()[0] instanceof DbQueryRunnerConn)) {
89                  throw new IllegalArgumentException(String.format(
90                          "First parameter must be a DbQueryRunnerConn instance, not %s",
91                          method.getName()));
92              }
93              // Get DbQueryRunnerConn object
94              final DbQueryRunnerConn dbAccess = (DbQueryRunnerConn) invocation.
95                      getArguments()[0];
96              // Begin transaction
97              dbAccess.getConnection().setAutoCommit(false);
98              try {
99                  // Proceed with the original method's invocation
100                 object = invocation.proceed();
101                 // Commit if successful
102                 dbAccess.getConnection().commit();
103                 dbAccess.getConnection().setAutoCommit(true);
104                 if (log.isDebugEnabled()) {
105                     log.debug(String.format(
106                             "Committed transaction for method %s",
107                             invocation.getMethod().getName()));
108                 }
109             } catch (Exception e) {
110                 // Rollback on Exception
111                 if (log.isDebugEnabled()) {
112                     log.debug(String.format(
113                             "Rollback transaction for method %s",
114                             invocation.getMethod().getName()));
115                 }
116                 // Rollback on error
117                 dbAccess.getConnection().rollback();
118                 dbAccess.getConnection().setAutoCommit(true);
119                 throw e;
120             }
121         }
122         return object;
123     }
