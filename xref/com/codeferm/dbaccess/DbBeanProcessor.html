7   package com.codeferm.dbaccess;
8   
9   import java.beans.PropertyDescriptor;
10  import java.sql.ResultSetMetaData;
11  import java.sql.SQLException;
12  import java.util.Arrays;
13  import java.util.Locale;
14  import org.apache.commons.dbutils.BeanProcessor;
15  
16  /**
17   * DbUtils {@code BeanProcessor} to convert underscore "_" database field names
18   * to camelCase.
19   *
20   *
21   * @see com.codeferm.dbaccess.DbQueryRunnerConn
22   * @see com.codeferm.dbaccess.DbQueryRunnerDs
23   *
24   * @author sgoldsmith
25   * @version 1.0.0
26   * @since 1.0.0
27   */
28  public class DbBeanProcessor extends BeanProcessor {
29  
30      /**
31       * Map ResultSet to bean properties.
32       *
33       * @param rsmd ResultSet to map.
34       * @param props Bean property descriptor.
35       * @return Array mapping field to bean.
36       * @throws SQLException Possible exception.
37       */
38      @Override
39      protected final int[] mapColumnsToProperties(final ResultSetMetaData rsmd,
40              final PropertyDescriptor[] props) throws SQLException {
41          final int cols = rsmd.getColumnCount();
42          final int[] columnToProperty = new int[cols + 1];
43          Arrays.fill(columnToProperty, PROPERTY_NOT_FOUND);
44          for (int propIndex = 0; propIndex < props.length; propIndex++) {
45              final String propName = props[propIndex].getName();
46              for (int col = 1; col <= cols; col++) {
47                  final String columnName = this.toCamelCase(rsmd.getColumnName(
48                          col));
49                  if (columnName.equalsIgnoreCase(propName)) {
50                      columnToProperty[col] = propIndex;
51                      break;
52                  }
53              }
54          }
55          return columnToProperty;
56      }
57  
58      /**
59       * Convert string with '_' to camelCase.
60       *
61       * @param underscoreStr Input string.
62       * @return camelCase String.
63       */
64      private String toCamelCase(final String underscoreStr) {
65          final StringBuffer sbStr = new StringBuffer();
66          final String[] str = underscoreStr.split("_");
67          boolean firstTime = true;
68          for (String temp : str) {
69              if (firstTime) {
70                  sbStr.append(temp.toLowerCase(Locale.ENGLISH));
71                  firstTime = false;
72              } else {
73                  sbStr.append(Character.toUpperCase(temp.charAt(0)));
74                  sbStr.append(temp.substring(1).toLowerCase());
75              }
76          }
77          return sbStr.toString();
78      }
79  }
