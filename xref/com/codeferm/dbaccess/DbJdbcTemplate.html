76      /**
77       * Return query results as list of beans. {@code Connection} closed
78       * automatically.
79       *
80       * @param <T> Type of object that the handler returns.
81       * @param sql SQL statement to execute.
82       * @param params Initialize the PreparedStatement's IN parameters.
83       * @param clazz {@code Class} to map results to.
84       * @return {@code List} of {@code <T>} typed objects.
85       */
86      @Override
87      public final <T> List<T> selectList(final String sql,
88              final Object[] params, final Class clazz) {
89          if (log.isDebugEnabled()) {
90              log.debug(String.format("selectBeanList: sql=%s, params=%s", sql,
91                      Arrays.asList(params)));
92          }
93          return template.query(sql, params,
94                  new BeanPropertyRowMapper(clazz));
95      }

133     /**
134      * Executes the given INSERT statement with parameter array and returns auto
135      * generate keys. JDBC driver needs to support RETURN_GENERATED_KEYS.
136      * {@code Connection} closed automatically.
137      *
138      * @param sql SQL statement to execute.
139      * @param params Initialize the PreparedStatement's IN parameters.
140      * @return Field name/value pairs of keys.
141      */
142     @Override
143     @SuppressFBWarnings(value = "SIC_INNER_SHOULD_BE_STATIC_ANON", justification
144             = "This mimics how Spring does it")
145     public final Map<String, Object> updateReturnKeys(final String sql,
146             final Object[] params) {
147         if (log.isDebugEnabled()) {
148             log.debug(String.format("updateReturnKeys: sql=%s, params=%s", sql,
149                     Arrays.asList(params)));
150         }
151         // Holds keys after update
152         final KeyHolder keyHolder = new GeneratedKeyHolder();
153         // Generic PreparedStatementCreator
154         final PreparedStatementCreator psc = new PreparedStatementCreator() {
155 
156             @Override
157             public final PreparedStatement createPreparedStatement(
158                     final Connection con) throws SQLException {
159                 PreparedStatement preparedStatement = con.
160                         prepareStatement(sql,
161                                 Statement.RETURN_GENERATED_KEYS);
162                 // Fill parameters
163                 int i = 1; //NOPMD
164                 for (Object o : params) {
165                     preparedStatement.setObject(i++, o);
166                 }
167                 return preparedStatement;
168             }
169         };
170         // Execute update
171         template.update(psc, keyHolder);
172         // Return keys
173         return keyHolder.getKeys();
174     }

176     /**
177      * Executes the given INSERT, UPDATE, or DELETE SQL statement with array of
178      * parameter arrays.
179      *
180      * @param sql SQL statement to execute.
181      * @param params Initialize the PreparedStatement's IN parameters.
182      * @return Number of rows updated array.
183      */
184     @Override
185     public final int[] batch(final String sql, final Object[][] params) {
186         if (log.isDebugEnabled()) {
187             log.debug(String.format("batch: sql=%s", sql));
188             for (Object[] param : params) {
189                 log.debug(
190                         String.format("batch: params=%s",
191                                 Arrays.asList(param)));
192             }
193         }
194         return template.batchUpdate(sql, Arrays.asList(params));
195     }
